The Tab component is a child of the Tabs component.

```jsx
import { useState } from 'react';
import { Tabs, Tab } from '@alaneicker/react-component-library';

<Tabs>
  <Tab label="Tab One" />
  <Tab label="Tab Two" />
  <Tab label="Tab Three" />
</Tabs>;
```
